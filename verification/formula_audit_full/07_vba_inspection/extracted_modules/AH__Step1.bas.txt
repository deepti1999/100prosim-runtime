Attribute VB_Name = "Step1"
Option Explicit

Public Sub Step1_ListWorkbooksAndSheets()

    Dim wb As Workbook
    Dim ws As Worksheet

    Debug.Print "===== STEP 1: OPEN WORKBOOKS & SHEETS ====="

    For Each wb In Application.Workbooks
        Debug.Print "Workbook: " & wb.Name
        For Each ws In wb.Worksheets
            Debug.Print "   Sheet: " & ws.Name
        Next ws
    Next wb

    Debug.Print "===== END STEP 1 ====="

End Sub
