Attribute VB_Name = "Trace_Snapshot"
Option Explicit

Private BeforeState As Object

' =========================
' SNAPSHOT BEFORE
' =========================
Public Sub SnapshotBefore_All()

    Dim wb As Workbook, ws As Worksheet, c As Range
    Set BeforeState = CreateObject("Scripting.Dictionary")

    For Each wb In Application.Workbooks
        For Each ws In wb.Worksheets
            For Each c In ws.UsedRange.Cells
                If c.HasFormula Then
                    On Error Resume Next
                    BeforeState(wb.Name & "|" & ws.Name & "|" & c.Address(False, False)) = _
                        Array(c.formula, c.Value)
                    On Error GoTo 0
                End If
            Next c
        Next ws
    Next wb

End Sub

' =========================
' SNAPSHOT AFTER
' =========================
Public Sub SnapshotAfter_All()

    Dim wb As Workbook, ws As Worksheet, c As Range
    Dim key As String
    Dim oldFormula As String, oldVal As Variant

    For Each wb In Application.Workbooks
        For Each ws In wb.Worksheets
            For Each c In ws.UsedRange.Cells
                If c.HasFormula Then

                    key = wb.Name & "|" & ws.Name & "|" & c.Address(False, False)

                    If BeforeState.Exists(key) Then
                        oldFormula = BeforeState(key)(0)
                        oldVal = BeforeState(key)(1)

                        ' ===== MODE A: ONLY CHANGES =====
                        If TRACE_MODE = "A" Then
                            If c.Value <> oldVal Then
                                LogLine wb.Name, ws.Name, c.Address(False, False), _
                                        c.formula, oldVal, c.Value, "CALC"
                            End If
                        End If

                        ' ===== MODE B: FULL STATE =====
                        If TRACE_MODE = "B" Then
                            LogLine wb.Name, ws.Name, c.Address(False, False), _
                                    c.formula, oldVal, c.Value, "CALC"
                        End If

                    End If

                End If
            Next c
        Next ws
    Next wb

End Sub
