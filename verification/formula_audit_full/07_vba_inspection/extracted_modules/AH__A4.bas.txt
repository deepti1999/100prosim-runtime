Attribute VB_Name = "A4"
Public Sub A4_2_Token_Based_Expansion_Test()

    Dim wb As Workbook
    Dim c As Range
    Dim tokens As Collection
    Dim t As Variant
    Dim r As Range

    Set wb = Workbooks("_S.xlsx")
    Set c = wb.Worksheets("0.Titel").Range("Z66")

    Debug.Print "===== A4.2 TOKEN-BASED EXPANSION ====="
    Debug.Print "Cell: " & c.Parent.Name & "!" & c.Address(False, False)
    Debug.Print "Formula: " & c.formula

    Set tokens = ExtractCellTokens(c.formula)

    For Each t In tokens
        On Error Resume Next
        Set r = c.Parent.Range(t)
        On Error GoTo 0

        If Not r Is Nothing Then
            Debug.Print "Resolved token -> " & r.Parent.Name & "!" & r.Address(False, False)
            Debug.Print "  Formula: " & r.formula
        Else
            Debug.Print "Could NOT resolve token: " & t
        End If
    Next t

    Debug.Print "===== END A4.2 ====="

End Sub

Private Function ExtractCellTokens(formula As String) As Collection
    ' Extracts tokens like A1, J66, AA123 from a formula

    Dim re As Object
    Dim matches As Object
    Dim m As Object
    Dim tokens As New Collection

    Set re = CreateObject("VBScript.RegExp")
    re.Pattern = "\b[A-Z]{1,3}[0-9]{1,7}\b"
    re.Global = True

    Set matches = re.Execute(formula)

    For Each m In matches
        tokens.Add m.Value
    Next m

    Set ExtractCellTokens = tokens
End Function

Public Sub A4_4_Resolve_INDIRECT_Runtime()

    Dim evalResult As Variant
    Dim resolvedRange As Range

    Set wb = Workbooks("_S.xlsx")
    Set ws = wb.Worksheets("2. Erneuerbare")

    Debug.Print "===== A4.4 RESOLVE INDIRECT (RUNTIME, FIXED) ====="

    ' ---------------- AI236 ----------------
    Set c = ws.Range("AI236")
    Debug.Print ""
    Debug.Print "Cell: 2. Erneuerbare!AI236"
    Debug.Print "Formula: " & c.formula

    evalResult = Application.Evaluate(c.formula)

    If TypeName(evalResult) = "Range" Then
        Set resolvedRange = evalResult
        Debug.Print "Resolved range: " & resolvedRange.Parent.Name & "!" & resolvedRange.Address

Public Sub A4_6_Resolve_INDIRECT_BindFixed()

    Dim wb As Workbook
    Dim ws As Worksheet
    Dim c As Range
    Dim addrText As String
    Dim sheetName As String
    Dim cellAddr As String
    Dim exclamPos As Long
    Dim resolvedRange As Range

    Set wb = Workbooks("_S.xlsx")
    Set ws = wb.Worksheets("2. Erneuerbare")

    Debug.Print "===== A4.6 RESOLVE INDIRECT (BIND FIXED) ====="

    ' -------- AI236 --------
    Set c = ws.Range("AI236")
    addrText = ws.Range("AG228").Value & ws.Range("AI228").Value & ws.Range("AG236").Value

    Debug.Print ""
    Debug.Print "AI236 resolved text: " & addrText

    exclamPos = InStr(addrText, "!")
    sheetName = Replace(Left(addrText, exclamPos - 1), "'", "")
    cellAddr = Mid(addrText, exclamPos + 1)

    Set resolvedRange = wb.Worksheets(sheetName).Range(cellAddr)
    Debug.Print "Resolved range: " & resolvedRange.Parent.Name & "!" & resolvedRange.Address

    ' -------- AH236 --------
    Set c = ws.Range("AH236")
    addrText = ws.Range("AG228").Value & ws.Range("AH228").Value & ws.Range("AG236").Value

    Debug.Print ""
    Debug.Print "AH236 resolved text: " & addrText

    exclamPos = InStr(addrText, "!")
    sheetName = Replace(Left(addrText, exclamPos - 1), "'", "")
    cellAddr = Mid(addrText, exclamPos + 1)

    Set resolvedRange = wb.Worksheets(sheetName).Range(cellAddr)
    Debug.Print "Resolved range: " & resolvedRange.Parent.Name & "!" & resolvedRange.Address

    Debug.Print ""
    Debug.Print "===== END A4.6 ====="

End Sub
