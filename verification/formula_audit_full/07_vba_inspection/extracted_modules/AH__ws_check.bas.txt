Attribute VB_Name = "ws_check"
Option Explicit

Dim WS_Before As Object

Public Sub SnapshotWS_Before()
    Dim wb As Workbook, ws As Worksheet, c As Range
    Set WS_Before = CreateObject("Scripting.Dictionary")

    For Each wb In Application.Workbooks
        For Each ws In wb.Worksheets
            If ws.Name Like "WS*" Then
                For Each c In ws.UsedRange
                    If c.HasFormula Or Not IsEmpty(c.Value) Then
                        WS_Before(wb.Name & "!" & ws.Name & "!" & c.Address) = c.Value
                    End If
                Next c
            End If
        Next ws
    Next wb

    Debug.Print "WS SNAPSHOT BEFORE taken."
End Sub


Public Sub SnapshotWS_After()
    Dim wb As Workbook, ws As Worksheet, c As Range
    Dim key As String, oldVal As Variant

    Debug.Print "=== WS CHANGES ==="

    For Each wb In Application.Workbooks
        For Each ws In wb.Worksheets
            If ws.Name Like "WS*" Then
                For Each c In ws.UsedRange
                    key = wb.Name & "!" & ws.Name & "!" & c.Address
                    If WS_Before.Exists(key) Then
                        oldVal = WS_Before(key)
                        If c.Value <> oldVal Then
                            Debug.Print key & " : " & oldVal & " -> " & c.Value
                        End If
                    End If
                Next c
            End If
        Next ws
    Next wb

    Debug.Print "=== END WS CHANGES ==="
End Sub
