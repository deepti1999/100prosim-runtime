Attribute VB_Name = "Modul2"
Sub EinzelModBasisRegen()
Attribute EinzelModBasisRegen.VB_ProcData.VB_Invoke_Func = " \n14"
'
' Archivoption Einzel-Modifikation: Regeneration Basisszenario und Ergänzung Basis-Historienspalte

' Erfolgreich getestet: hsk 24.03.2021
'

    Dim Quelle As String, Ziel As String
    Quelle = "_S.Basis.xlsx"
    Ziel = "_S.xlsx"
    
    If Range("Archivoption").Value <> 2 Then
        Exit Sub
    End If
    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    
    Workbooks(Ziel).Close
    
    FileCopy Quelle, Ziel

    Workbooks.Open (Ziel)
    SendKeys "{ENTER}", 2
  
    Application.DisplayAlerts = False
    Call AllesAbgleichen
    Call ArchivCockpit2Basis
    Application.DisplayAlerts = True
    Application.ScreenUpdating = True
End Sub

Sub ArchivCockpit2Basis()
'
' ArchivCockpit2Basis, Delta zu ArcivCockpit2: ohne Eingangsabfrage, ohne Änderungsmarker
'
'
Application.ScreenUpdating = False
    Range("CP2SzenKenn") = "Basis"
    SzenStandKenn = Range("CP2SzenKenn").Value
    Range("CP2SzenKenn").ClearContents
    Sheets("Monitor").Unprotect Password:="m"
    Range("MonSzenKenn").Value = SzenStandKenn
    Call Zielvariante
'    Call Aenderungsmarker
    Sheets("Monitor").Protect Password:="m"
    
    Call Historienspeicher
    
    Range("D3").Select
    Sheets("Cockpit2").Select
    Range("Archiviert").Value = 1
    Range("D2").Select
    Application.ScreenUpdating = True
End Sub
