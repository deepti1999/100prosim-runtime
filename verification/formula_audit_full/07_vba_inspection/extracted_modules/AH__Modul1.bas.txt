Sub AllesAbgleichen()
Attribute AllesAbgleichen.VB_ProcData.VB_Invoke_Func = "a\n14"

' AllesAbgleichen Makro (wirkt auf Solar-Freiflächen)

'#Stefan Sonntag - Test und Verifikation: OK (hsk 24.03.2021)
    
    Application.ScreenUpdating = False

    Call TraktionVerkehrsleistung
    Call HolzGebaeudewaerme

    Atime = Timer()

    For Durchlauf = 1 To 10
        Call PW_Brennstoff
        Call SolarFreiflaeche
        Call GWaermeDirekt
        Call JahresbilanzStromAbgleich

        If Range("Sum") = 0 Then
            Exit For
        End If
    Next Durchlauf

    Btime = Timer()

    Workbooks("AH.xlsm").Activate
    Sheets("Cockpit2").Select
    Range("Archiviert").Value = 0

    Warnung = ""

    Application.ScreenUpdating = True

    If Range("SolFrei") < 0 Then
        Warnung = "   ACHTUNG! Negative Solar-Freifläche wegen Überangebot. Andere Energiebeiträge reduzieren."
    End If

    MsgBox ("Der Abgleich wurde in " & Durchlauf & " Durchläufen berechnet in " & Format(Btime - Atime, ".00") & " Sekunden" & Warnung)

End Sub

Sub ArchivCockpit2()
'
' ArchivCockpit2 Makro
'
'
Application.ScreenUpdating = False
    If Range("Archiviert").Value > 0 Then
        MsgBox "Archivierung nicht sinnvoll - kein neuer Abgleich seit der letzten Archivierung"
        Range("Archiviert").Value = 2   'Unterscheidung zu 1 am Ende des normalen Durchlaufs
        Exit Sub
    End If
    If Range("CP2SzenKenn").Value = "" Then
        MsgBox "Zuerst Kennung eintragen!"
        Range("Archiviert").Value = 2   'Unterscheidung zu 1 am Ende des normalen Durchlaufs
        Exit Sub
    End If
    SzenStandKenn = Range("CP2SzenKenn").Value
    Range("CP2SzenKenn").ClearContents
    Sheets("Monitor").Unprotect Password:="m"
    Range("MonSzenKenn").Value = SzenStandKenn
    Call Zielvariante
    Call Aenderungsmarker
    Sheets("Monitor").Protect Password:="m"

    Call Historienspeicher

    Range("D3").Select
    Sheets("Cockpit2").Select
    Range("Archiviert").Value = 1
    Range("D2").Select
    Application.ScreenUpdating = True

    Call EinzelModBasisRegen
End Sub

Sub Zielvariante()
'
' Zielvariante Makro
' Speichert die aktuellen Zieldaten in AH.Monitor aus S als Zieldaten der vorhergehenden Variante ab und eröffnet so die nächste Variation
'
'
    Sheets("Monitor").Select
    Columns("X:X").Select
    Selection.Copy
'    Application.CutCopyMode = xlCopy 'weder im Kopiermodus (xlCopy) noch im Ausschneidemodus (xlCut)
    Columns("AN:AN").Select
    Selection.Insert Shift:=xlToRight
    Selection.PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks _
        :=False, Transpose:=False 'Als Selection.Insert möglich?

    Columns("AL:AL").Select
    Selection.Copy
    Columns("AN:AN").Select
    Selection.PasteSpecial Paste:=xlPasteFormats, Operation:=xlNone, _
        SkipBlanks:=False, Transpose:=False
    Selection.EntireColumn.Hidden = False

'    Application.CutCopyMode = False
    Range("AC6").Select
    Application.CutCopyMode = False
    Selection.Copy
    Range("M1").Select
    Selection.PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks _
        :=False, Transpose:=False 'Kann nicht einfügen, da nicht die gleiche Größe

    Range("X5").Select
    Selection.Copy
    Range("NrZielHistorie").Select
    Selection.PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks _
        :=False, Transpose:=False
    Range("D4").Select
End Sub

Sub Aenderungsmarker()

' Markiert die Zellen des zuletzt archivierten Standes, die sich gegenüber dem Vorzustand unterscheiden

    Sheets("Monitor").Select
    Start = Range("U3").Value
    Ende = Range("U4").Value
    For Zeile = Start To Ende
        If Range("AN" & Zeile).Value <> Range("AO" & Zeile) Then
            Range("AN" & Zeile).Select
            Selection.Font.Bold = True
            With Selection.Interior
                .Pattern = xlSolid
                .PatternColorIndex = xlAutomatic
                .ThemeColor = xlThemeColorAccent5
'       .TintAndShade = -0.249977111117893
        .TintAndShade = 0.799981688894314
                .PatternTintAndShade = 0
            End With
        End If
    Next Zeile
End Sub
Sub HistorieAnfang()

' Historienspeicher unter MH....xlsx sichern, löschen und auf Anfang zurücksetzen: Status und Basis-Szenario
' Erfolgreich getestet 22.04.2020 0:45
' 30.07.2020 kopierten Spaltenbereich von AA auf BZ ausgeweitet

Select Case MsgBox("Historienspeicher tatsächlich löschen und in Anfangszustand bringen?", _
                vbYesNo Or vbQuestion Or vbDefaultButton1, "Hinweis")
    Case vbYes
        Application.ScreenUpdating = False
' Arbeitsblatt-Inhalt AH.Historie wird nach MH.Modifikations-Historie kopiert und mit Datumskennung gespeichert.
    Call HistorieSpeichern


' AH.Monitor - Historienteil löschen und Status/Basis-Zielspalten als Start in AK:AL und AN:AO aufnehmen
        Workbooks("AH.xlsm").Activate
        Sheets("Monitor").Select
        Sheets("Monitor").Unprotect Password:="m"

        Range("AE5").Value = 0
        Columns("AN:BK").Select
        Selection.Delete Shift:=xlToLeft
        Columns("W:X").Select
        Selection.Copy
        Columns("AK:AL").Select
        Selection.PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks _
            :=False, Transpose:=False
        Range("AL6").Value = "Basis"
        Application.CutCopyMode = xlCopy 'weder im Kopiermodus (xlCopy) noch im Ausschneidemodus (xlCut)
        Columns("AK:AK").Select
        Selection.Copy
        Columns("AN:AN").Select

        Selection.Insert Shift:=xlToRight
        Selection.EntireColumn.Hidden = False
        Columns("AL:AL").Select
        Selection.Copy
        Columns("AN:AN").Select

        Selection.Insert Shift:=xlToRight
        Selection.EntireColumn.Hidden = False
        Call Aenderungsmarker
        Range("X5").Select
        Selection.Copy
        Range("NrZielHistorie").Select
        Selection.PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks _
            :=False, Transpose:=False
        Range("D4").Select
        Sheets("Monitor").Protect Password:="m"
        Range("P7").Select
' Arbeitsblatt AH.Historie komplett löschen
        Sheets("Historie").Unprotect
        Sheets("Historie").Range("A:YZ").Delete Shift:=xlLeft
'        Columns("A:YZ").Select
 '       Selection.Delete Shift:=xlToLeft
'Bezeichungsspalten links aus AH.Monitor kopieren und in AH.Historie einfügen.
        Sheets("Monitor").Select
        Columns("A:M").Select
        Range("M1").Activate
        Selection.Copy
        Sheets("Historie").Select
        Range("A1").Select
        ActiveSheet.Paste
'Formatierte Statusspalte AK:AK nach Historie N:N kopieren, Basis-Zielspalte AL:AL vor N:N einfügen
        Worksheets("Monitor").Range("AN:AO").Copy
        Worksheets("Historie").Range("N:O").Insert Shift:=xlToRight
        Sheets("Historie").Protect
        Application.ScreenUpdating = True
        Worksheets("0.Cockpit").Select
    Case vbNo
    End Select

End Sub

Sub Historienspeicher()
'
' Fügt die jüngste Modifikation in AH.Monitor aus AN:AN in AH.Historie N:N ein.
'
'
    Worksheets("Historie").Unprotect
    Worksheets("Monitor").Range("AN:AN").Copy
    Worksheets("Historie").Range("N:N").Insert Shift:=xlToRight
    Titel = Worksheets("Monitor").Range("G5") 'Diagrammtitel übernehmen
    Worksheets("Historie").Range("G5") = Titel 'und in Historie aktualisieren
    Worksheets("Historie").Protect DrawingObjects:=True, Contents:=True, Scenarios:=True

End Sub

Sub HistorieSpeichern()

' Historienspeicher unter MH....xlsx sichern
' Erfolgreich getestet 22.04.2020 0:45
' 30.07.2020 kopierten Spaltenbereich von AA auf BZ ausgeweitet

' Arbeitsblatt-Inhalt AH.Historie wird nach MH.Modifikations-Historie kopiert und mit Datumskennung gespeichert.
    Dateipfad = Range("DateiMHsave")
    DateipfadLoad = Range("DateiMHload")
    Workbooks("AH.xlsm").Sheets("Historie").Range("A:BZ").Copy '30.07.2020 kopierten Spaltenbereich von AA auf BZ ausgeweitet.
    Workbooks.Open Filename:=DateipfadLoad
    Workbooks("MH.xlsx").Sheets("Modifikations-Historie").Paste
    Workbooks("MH.xlsx").SaveAs Filename:=Dateipfad
    Rows("2:3").Select
    Selection.EntireRow.Hidden = True
    Range("D5").Select
    ActiveSheet.Protect DrawingObjects:=True, Contents:=True, Scenarios:=True
End Sub
