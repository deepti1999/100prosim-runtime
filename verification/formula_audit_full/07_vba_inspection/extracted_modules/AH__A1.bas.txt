Attribute VB_Name = "A1"
Option Explicit

Public Sub A1_Show_sZiel_and_sMod()

    Dim wb As Workbook
    Dim rZiel As Range, rMod As Range

    ' --- explicitly reference the correct workbook ---
    Set wb = Workbooks("_S.xlsx")

    ' s_Ziel
    Set rZiel = wb.Names("s_Ziel").RefersToRange

    Debug.Print "===== A1: s_Ziel ====="
    Debug.Print "Workbook: " & wb.Name
    Debug.Print "Sheet: " & rZiel.Parent.Name
    Debug.Print "Address: " & rZiel.Address(False, False)
    Debug.Print "Formula: " & rZiel.formula
    Debug.Print "Value: " & rZiel.Value

    ' s_Mod
    Set rMod = wb.Names("s_Mod").RefersToRange

    Debug.Print "===== A1: s_Mod ====="
    Debug.Print "Workbook: " & wb.Name
    Debug.Print "Sheet: " & rMod.Parent.Name
    Debug.Print "Address: " & rMod.Address(False, False)
    Debug.Print "Formula: " & rMod.formula
    Debug.Print "Value: " & rMod.Value

    Debug.Print "===== END A1 ====="

End Sub

Public Sub A2_Show_KLIK()

    Dim wb As Workbook
    Dim rKLIK As Range

    Set wb = Workbooks("_S.xlsx")

    ' KLIK may be a named range
    On Error Resume Next
    Set rKLIK = wb.Names("KLIK").RefersToRange
    On Error GoTo 0

    If rKLIK Is Nothing Then
        Debug.Print "KLIK is NOT a named range. It is likely a cell formula reference."
        Debug.Print "Search manually: look at the precedent of s_Ziel (U13)."
        Exit Sub
    End If

    Debug.Print "===== A2: KLIK ====="
    Debug.Print "Workbook: " & wb.Name
    Debug.Print "Sheet: " & rKLIK.Parent.Name
    Debug.Print "Address: " & rKLIK.Address(False, False)
    Debug.Print "Formula: " & rKLIK.formula
    Debug.Print "Value: " & rKLIK.Value
    Debug.Print "===== END A2 ====="

End Sub
